# 🎬 iniciador.py — O Maestro da Pipeline
# Orquestra os 3 estágios em sequência

from extracao import extrair_csv, categorizar_usuarios
from transformacao import transformar_usuarios
from carregamento import carregar_em_markdown

def executar_pipeline():
    """Pipeline ETL completo com Gemini AI"""
    
    # ── EXTRACT ───────────────────────────────
    df = extrair_csv("usuarios.csv")
    categorizados = categorizar_usuarios(df)
    
    # ── TRANSFORM (Gemini faz a magia aqui) ──
    transformados = transformar_usuarios(categorizados)
    
    # ── LOAD ──────────────────────────────────
    carregar_em_markdown(transformados)
    
    print(f"✅ Pipeline completo! {len(transformados)} usuários")

if __name__ == "__main__":
    executar_pipeline()
# 📥 extracao.py — Lê e Organiza Dados

import pandas as pd
from pathlib import Path

def extrair_csv(nome_arquivo: str) -> pd.DataFrame:
    """Extrai dados do CSV e retorna DataFrame"""
    caminho = Path("../dados") / nome_arquivo
    df = pd.read_csv(caminho)
    print(f"[EXTRACT] {len(df)} usuários carregados")
    return df

def categorizar_usuarios(df: pd.DataFrame) -> dict:
    """Separa usuários por departamento"""
    departamentos = df['departamento'].unique()
    
    return {
        dept: df[df['departamento'] == dept].to_dict('records')
        for dept in departamentos
    }
# 🤖 transformacao.py — Gemini Fazendo Magia

import google.generativeai as genai
import os

genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
model = genai.GenerativeModel("gemini-pro")

def gerar_resumo(usuario: dict) -> str:
    """Gemini gera um resumo criativo do usuário"""
    prompt = f"""
    Crie um resumo descontraído e criativo sobre:
    Nome: {usuario['nome']}
    Empresa: {usuario['empresa']}  
    Cidade: {usuario['cidade']}
    Departamento: {usuario['departamento']}
    
    Seja criativo, use emojis, max 3 parágrafos.
    """
    response = model.generate_content(prompt)
    return response.text

def transformar_usuarios(categorizados: dict) -> list:
    """Aplica IA em cada usuário"""
    resultado = []
    for dept, usuarios in categorizados.items():
        for usuario in usuarios:
            usuario['resumo_ia'] = gerar_resumo(usuario)
            resultado.append(usuario)
    return resultado
# 💾 carregamento.py — Salva em Markdown

from pathlib import Path

def gerar_markdown(usuario: dict) -> str:
    """Formata usuário como documento Markdown"""
    return f"""# {usuario['nome']}

## 📋 Informações Básicas
- **Email**: {usuario['email']}
- **Cidade**: {usuario['cidade']}
- **Empresa**: {usuario['empresa']}
- **Departamento**: {usuario['departamento']}

## 🤖 Resumo Gerado por IA
{usuario['resumo_ia']}

---
*Gerado por Pipeline ETL × Gemini AI*
"""

def carregar_em_markdown(usuarios: list):
    """Salva cada usuário como arquivo .md"""
    output = Path("resumos_markdown")
    output.mkdir(exist_ok=True)
    
    for u in usuarios:
        nome_arquivo = u['nome'].replace(" ", "_")
        arquivo = output / f"{nome_arquivo}.md"
        arquivo.write_text(gerar_markdown(u))
    
    print(f"[LOAD] {len(usuarios)} arquivos salvos")
